# Load ggplot2
library(ggplot2)

# Load the data
data <- data.frame(
  Groups=c("Female 1970", "Male 1970", "Female 2020", "Male 2020") , 
  Enrollment=c(8.2, 14.1, 38.3, 36.7), 
  Bachelors=c(31, 42, 92, 67)
)

# Calculate the future positions on the x axis of each bar (left border, central position, right border)

data$right <- cumsum(data$Bachelors) + 30*c(0:(nrow(data)-1))
data$left <- data$right - data$Bachelors

# Plot
ggplot(data, aes(ymin = 0)) + 
  geom_rect(aes(xmin = left, xmax = right, ymax = Enrollment, colour = Groups, fill = Groups)) +
  ggtitle("College Enrollment and Percentage with \nat Least a Bachelor's by Year and Gender") +
  xlab("Enrollment by Year and Gender") + 
  ylab("Percentage with at Least a Bachelor's") +
  theme(plot.title = element_text(size = 12, hjust = 0.5),
        axis.title.y = element_text(size = 12),
        axis.title.x = element_text(size = 12))
